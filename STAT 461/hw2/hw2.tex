\section{2.4.2}
\begin{flalign}
P(A) &= 0.2 \\
P(B) &= 0.4 \\
P(A|B) + P(B|A) &= 0.75
\end{flalign}
Find:
\begin{flalign}
P(A \cap B)
\end{flalign}
\begin{flalign}
P(A|B) + P(B|A) &= 0.75 \\
\frac{P(A \cap B)}{P(B)} + \frac{P(A \cap B)}{P(A)} &= 0.75 \\
P(A \cap B) &= \frac{0.75}{\frac{1}{P(A)} + \frac{1}{P(B)}} \\
&= \frac{0.75}{\frac{1}{0.2} + \frac{1}{0.4}} \\
P(A \cap B) &= 0.1
\end{flalign}

P(A^c \cap B^c)

% problem 2.5.7
\section{2.5.7}
\begin{flalign}
P(A) &= \frac{1}{4} \\
P(B) &= \frac{1}{8}
\end{flalign}

\subsection{}
A, B are mutually exclusive
\begin{flalign}
P(A \cup B) &= P(A) + P(B) \\
&= \frac{1}{4} + \frac{1}{8} \\
P(A \cup B) &= \frac{3}{8}
\end{flalign}

\subsection{}
A, B are independent
\begin{flalign}
P(A \cup B) &= 1 - P(A^c \ cap B^c) \\
&= 1 - P(A^c)P(B^c) \\
&= 1 - (1 - \frac{1}{4})(1 - \frac{1}{8}) \\
P(A \cup B) &= \frac{11}{32}
\end{flalign}

\subsection{}
A, B are mutually exclusive
\begin{flalign}
P(A | B) &= \frac{P(A \cap B)}{P(B)} \\
&= \frac{0}{P(B)} \\
P(A | B) &= 0
\end{flalign}

\subsection{}
A, B are independent
\begin{flalign}
P(A | B) &= \frac{P(A \cap B)}{P(B)} \\
&= \frac{P(A)P(B)}{P(B)} \\
&= P(A) \\
&= \frac{1}{4}
\end{flalign}
